 184     <xs:complexType name="MessageHeadersType">
 185         <xs:sequence>
 186             <xs:element name="MessageId" type="UuidType">
 187                 <xs:annotation>
 188                     <xs:documentation>
 189                         Globally unique identifier of the current message. It's the responsibility
 190                         of the origin system to generate this as a valid UUID (Universally
 191                         unique identifier)
 192                     </xs:documentation>
 193                 </xs:annotation>
 194             </xs:element>
 195             <xs:element name="CreationDateTime" type="xs:dateTime">
 196                 <xs:annotation>
 197                     <xs:documentation>
 198                         Timestamp of when this message was constructed. This value must be
 199                         consistent (specially in the case of clusters) and precise (include
 200                         milliseconds and timezone). For example: 2012-07-16T19:20:30.450+01:00
 201                     </xs:documentation>
 202                 </xs:annotation>
 203             </xs:element>
 204             <xs:element name="CorrelationId" type="xs:string" minOccurs="0">
 205                 <xs:annotation>
 206                     <xs:documentation>
 207                         Generic identifier that provides correlation between messages. The
 208                         format of this ID is context dependent.
 209                     </xs:documentation>
 210                 </xs:annotation>
 211             </xs:element>
 212             <xs:element name="SystemInfo" type="xs:string" minOccurs="0">
 213                 <xs:annotation>
 214                     <xs:documentation>
 215                         String that identifies the system sending the message. It's used
 216                         for auditing purposes only.
 217                     </xs:documentation>
 218                 </xs:annotation>
 219             </xs:element>
 220             <xs:element name="MessageClass" type="NonBlankStringType" minOccurs="0">
 221                 <xs:annotation>
 222                     <xs:documentation>
 223                         This field is not populated for normal XML messages. It exists in the schema only to support the translation
 224                         to other formats like JSON that do not provide a way for the receiver to determine the message class from
 225                         the payload. For all other uses it should be ignored.
 226                     </xs:documentation>
 227                 </xs:annotation>
 228             </xs:element>
 229         </xs:sequence>
