Void

Example
String fieldValue = currentDocument.getDocumentFieldValue("FIELD001");
if (fieldValue == "VALUE1") {
  supportingDoc.addAnyDocuments(["Document001", "Spouse"], ["Document002", "Spouse"]);
} else {
  returnList.get(0).setScriptReturnErrorString("Document needs review");
}
